    ```tsx
    // src/components/design/TrustBadge.tsx
    import type { JSX } from "react";

    type Tier = "good" | "warn" | "bad";

    function tierFor(score: number): Tier {
      if (score >= 90) return "good";
      if (score >= 70) return "warn";
      return "bad";
    }

    export interface TrustBadgeProps {
      score: number;
      onClick?: () => void;
      size?: "sm";  // only sm in v1 per D-08
    }

    export function TrustBadge({ score, onClick, size = "sm" }: TrustBadgeProps): JSX.Element {
      const tier = tierFor(score);
      return (
        <span
          className={`trust ${tier}`}
          data-size={size}
          onClick={(e) => {
            e.stopPropagation();
            onClick?.();
          }}
          role={onClick ? "button" : undefined}
        >
          <span className="trust-dot" /> {score}%
        </span>
      );
    }
    ```

    **TrustBadge port details:**
    - Tier formula EXACT per D-08 + prototype line 41: `>= 90 → "good"`, `>= 70 → "warn"`, otherwise `"bad"`. Do NOT generalise these thresholds to props or env — RESEARCH.md §"Specifics" says they're product semantics, not styling.
    - `tierFor` extracted as a pure helper (improves testability + readability vs. inline ternary).
    - `data-size={size}` attribute is a hook for future size variants without changing the className shape (v1 only ships `sm`).
    - `onClick` invokes `e.stopPropagation()` BEFORE calling `onClick?.()` — verbatim from prototype line 43.
    - `role="button"` only when `onClick` is provided — without onClick, the badge is decorative, not interactive (a11y nicety beyond the prototype).
    - Type `Tier` is module-private; only `TrustBadgeProps` and `TrustBadge` are exported.
